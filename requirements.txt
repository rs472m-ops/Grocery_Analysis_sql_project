-- ============================================
-- GROCERY STORE MANAGEMENT - ANALYSIS QUERIES
-- ============================================

-- ================================
-- 1. CUSTOMER INSIGHTS
-- ================================

-- Total unique customers who placed orders
SELECT COUNT(DISTINCT cust_id) AS total_customers
FROM orders;

-- Customers with highest number of orders
SELECT c.cust_name, COUNT(o.ord_id) AS total_orders
FROM customers c
JOIN orders o ON c.cust_id = o.cust_id
GROUP BY c.cust_name
ORDER BY total_orders DESC;

-- Total and average purchase per customer
SELECT c.cust_name,
SUM(od.total_price) AS total_spent,
AVG(od.total_price) AS avg_spent
FROM customers c
JOIN orders o ON c.cust_id = o.cust_id
JOIN order_details od ON o.ord_id = od.ord_id
GROUP BY c.cust_name;

-- Top 5 customers by spending
SELECT c.cust_name,
SUM(od.total_price) AS total_spent
FROM customers c
JOIN orders o ON c.cust_id = o.cust_id
JOIN order_details od ON o.ord_id = od.ord_id
GROUP BY c.cust_name
ORDER BY total_spent DESC
LIMIT 5;

-- ================================
-- 2. PRODUCT PERFORMANCE
-- ================================

-- Number of products in each category
SELECT cat.cat_name, COUNT(p.prod_id) AS total_products
FROM categories cat
LEFT JOIN products p ON cat.cat_id = p.cat_id
GROUP BY cat.cat_name;

-- Average price of products by category
SELECT cat.cat_name, AVG(p.price) AS avg_price
FROM categories cat
JOIN products p ON cat.cat_id = p.cat_id
GROUP BY cat.cat_name;

-- Products with highest sales volume (quantity)
SELECT p.prod_name, SUM(od.quantity) AS total_quantity
FROM products p
JOIN order_details od ON p.prod_id = od.prod_id
GROUP BY p.prod_name
ORDER BY total_quantity DESC;

-- Total revenue generated by each product
SELECT p.prod_name, SUM(od.total_price) AS total_revenue
FROM products p
JOIN order_details od ON p.prod_id = od.prod_id
GROUP BY p.prod_name
ORDER BY total_revenue DESC;

-- Product sales by category and supplier
SELECT cat.cat_name, s.sup_name,
SUM(od.total_price) AS total_sales
FROM order_details od
JOIN products p ON od.prod_id = p.prod_id
JOIN categories cat ON p.cat_id = cat.cat_id
JOIN supplier s ON p.sup_id = s.sup_id
GROUP BY cat.cat_name, s.sup_name
ORDER BY total_sales DESC;

-- ================================
-- 3. SALES & ORDER TRENDS
-- ================================

-- Total number of orders
SELECT COUNT(*) AS total_orders
FROM orders;

-- Average order value
SELECT AVG(order_total) AS avg_order_value
FROM (
SELECT ord_id, SUM(total_price) AS order_total
FROM order_details
GROUP BY ord_id
) t;

-- Dates with most orders
SELECT order_date, COUNT(*) AS total_orders
FROM orders
GROUP BY order_date
ORDER BY total_orders DESC;

-- Monthly trends (order volume & revenue)
SELECT
MONTH(STR_TO_DATE(o.order_date, '%Y-%m-%d')) AS month,
COUNT(DISTINCT o.ord_id) AS total_orders,
SUM(od.total_price) AS total_revenue
FROM orders o
JOIN order_details od ON o.ord_id = od.ord_id
GROUP BY month
ORDER BY month;

-- Weekday vs Weekend orders
SELECT
CASE
WHEN DAYOFWEEK(STR_TO_DATE(order_date, '%Y-%m-%d')) IN (1,7)
THEN 'Weekend'
ELSE 'Weekday'
END AS day_type,
COUNT(*) AS total_orders
FROM orders
GROUP BY day_type;

-- ================================
-- 4. SUPPLIER CONTRIBUTION
-- ================================

-- Total suppliers
SELECT COUNT(*) AS total_suppliers
FROM supplier;

-- Supplier providing most products
SELECT s.sup_name, COUNT(p.prod_id) AS total_products
FROM supplier s
JOIN products p ON s.sup_id = p.sup_id
GROUP BY s.sup_name
ORDER BY total_products DESC;

-- Average price of products by supplier
SELECT s.sup_name, AVG(p.price) AS avg_price
FROM supplier s
JOIN products p ON s.sup_id = p.sup_id
GROUP BY s.sup_name;

-- Supplier contribution to revenue
SELECT s.sup_name, SUM(od.total_price) AS total_revenue
FROM supplier s
JOIN products p ON s.sup_id = p.sup_id
JOIN order_details od ON p.prod_id = od.prod_id
GROUP BY s.sup_name
ORDER BY total_revenue DESC;

-- ================================
-- 5. EMPLOYEE PERFORMANCE
-- ================================

-- Number of employees handling orders
SELECT COUNT(DISTINCT emp_id) AS total_employees
FROM orders;

-- Employees with most orders
SELECT e.emp_name, COUNT(o.ord_id) AS total_orders
FROM employees e
JOIN orders o ON e.emp_id = o.emp_id
GROUP BY e.emp_name
ORDER BY total_orders DESC;

-- Total sales handled by each employee
SELECT e.emp_name, SUM(od.total_price) AS total_sales
FROM employees e
JOIN orders o ON e.emp_id = o.emp_id
JOIN order_details od ON o.ord_id = od.ord_id
GROUP BY e.emp_name;

-- Average order value per employee
SELECT e.emp_name,
AVG(order_total) AS avg_order_value
FROM employees e
JOIN (
SELECT o.emp_id, od.ord_id, SUM(od.total_price) AS order_total
FROM orders o
JOIN order_details od ON o.ord_id = od.ord_id
GROUP BY o.emp_id, od.ord_id
) t ON e.emp_id = t.emp_id
GROUP BY e.emp_name;

-- ================================
-- 6. ORDER DETAILS ANALYSIS
-- ================================

-- Relationship between quantity and total price
SELECT quantity, AVG(total_price) AS avg_total_price
FROM order_details
GROUP BY quantity;

-- Average quantity ordered per product
SELECT p.prod_name, AVG(od.quantity) AS avg_quantity
FROM products p
JOIN order_details od ON p.prod_id = od.prod_id
GROUP BY p.prod_name;

-- Unit price variation across products
SELECT prod_id,
MIN(each_price) AS min_price,
MAX(each_price) AS max_price
FROM order_details
GROUP BY prod_id;
